\tikzset{
    cone/.style={
        ultra thick,
    }
}
\pgfmathsetmacro{\l}{5.1} %circumference of the marble
\newcommand{\cone}[1]{
    \pgfmathsetmacro{\a}{#1}
    \pgfmathsetmacro{\ri}{\l/\a}
    \draw[ultra thick] (0,0) circle [radius=0.225\textwidth];
    \draw[ultra thick] (\ri,0)--(0.225\textwidth,0);
    \draw[ultra thick, dashed] ({deg(#1)}:\ri)--({deg(#1)}:0.225\textwidth);
    \draw[ultra thick] (0,0) circle [radius=\ri];
}
\begin{document}
\pgfmathtruncatemacro{\h}{\the\textheight/(0.47*\the\textwidth)}
\pgfmathsetmacro{\n}{2*\h}
\foreach \angle in {pi,4*pi/3,3*pi/2}{
\begin{tikzpicture}
    \foreach \n in {1,...,\n} {
    \pgfmathtruncatemacro{\row}{int((\n-1)/2)}
    \pgfmathtruncatemacro{\col}{mod(\n-1,2)}
    \begin{scope}[
        xshift=\col*0.475\textwidth,
        yshift=-\row*0.47\textwidth
    ]
        \cone{\angle}
    \end{scope}
}
\end{tikzpicture}
\newpage    
}
\end{document}
